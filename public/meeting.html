'm=' + section : section; }); }, getMid(section) { const midMatch = section.match(/\na=mid:(.*)/); return midMatch ? midMatch[1] : null; }, reorderSDP(sdp, order) { const sections = this.getMediaSections(sdp); const header = sections[0]; const media = sections.slice(1); // Create ordered media sections based on the provided order const orderedMedia = order.map(mid => { return media.find(section => this.getMid(section) === mid); }).filter(Boolean); return [header, ...orderedMedia].join('\n'); } }; socket.on('offer', async ({ offer, from, fromUser }) => { console.log('📤 Received offer from:', fromUser || from); try { let peerConnection = peerConnections.get(from);
